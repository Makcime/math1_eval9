\question[3]
\noindent\textbf{TRACE} un triangle ABC tel que
$AB = \versions{4}{5}{6}{7}\,\text{cm}$, $AC = \versions{6}{7}{4}{5}\,\text{cm}$ \\
et lâ€™angle $\widehat{BAC} = \versions{40}{50}{60}{70}^\circ$.

\ifcorr{
	\vspace{0.8em}
	\noindent\textbf{Correction :} construction possible.

	\begin{center}
		\begin{tikzpicture}[scale=1]
			\color{red}
			% Longueurs et angle choisis par version
			\versions{
				\def\AB{4}
				\def\AC{6}
				\def\ang{40}
			}{
				\def\AB{5}
				\def\AC{7}
				\def\ang{50}
			}{
				\def\AB{6}
				\def\AC{4}
				\def\ang{60}
			}{
				\def\AB{7}
				\def\AC{5}
				\def\ang{70}
			}

			\coordinate (A) at (0,0);
			\coordinate (B) at (\AB,0);
			\coordinate (C) at ({\AC*cos(\ang)},{\AC*sin(\ang)});

			\draw[thick] (A) -- (B) -- (C) -- cycle;

			% Marquage de l'angle en A
			\draw (A) ++(0.9,0) arc (0:\ang:0.9);

			\node[below] at (A) {$A$};
			\node[below] at (B) {$B$};
			\node[above] at (C) {$C$};
		\end{tikzpicture}
	\end{center}
}
